# ── 4. OCR engines — scanned / image-only PDFs ───────────────────
#
# Install at least ONE of these for scanned PDFs to work.
# Priority (fastest → most capable):
#   Tesseract → PaddleOCR → TrOCR
#
# The parsers auto-detect which engines are installed at startup
# and skip any that are missing — no crashes, no unnecessary fallbacks.

# Option A: Tesseract (RECOMMENDED — fastest, ~5s/10 pages)
#   Linux:   sudo apt install tesseract-ocr
#   macOS:   brew install tesseract
#   Windows: https://github.com/UB-Mannheim/tesseract/wiki
pytesseract
opencv-python

# Option B: PaddleOCR (better for tables and complex layouts)
#   Fix for OneDNN crash on Windows: use exactly 2.7.3
paddleocr==2.7.3
paddlepaddle

# Option C: TrOCR via HuggingFace (last resort — slow, GPU recommended)
#   torch and transformers are also required by the LLM evaluator (see §5).
torch
torchvision
transformers


# ── 5. AI / LLM ───────────────────────────────────────────────────

# Optional: leave uninstalled if you only use Groq
anthropic
